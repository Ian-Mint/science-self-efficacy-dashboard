15def get_feature_importance(y, fields, file_loc=config.student_data_file, method='pearson'): 

16 """ 

17 Computes feature importance of all the variables in fields parameter with dependent y variable. For categorical 

18 fields, it provides results from anova analysis and for numerical/continuous fields it returns correlation 

19 coefficients with y. 

20  

21 :param method: Method of correlation 

22 - pearson : standard correlation coefficient 

23 - kendall : Kendall Tau correlation coefficient 

24 - spearman : Spearman rank correlation 

25 :type method: str 

26 :param file_loc: path to dataset 

27 :type file_loc: str 

28 :param fields: list of field ids where each field id is a string 

29 :type fields: list 

30 :param y: a dependent y field id 

31 :type y: str 

32 :returns: a multilevel dictionary, value corresponding to key 'category' contains a dictionary with anova results for 

33 categorical fields and value for key 'continuous' is dictionary with correlation coefficients. 

34 :rtype dict 

35 """ 

36 

37 df = load_data_frame(file_loc) 

38 var_info = get_var_info() 

39 

40 assert isinstance(fields, list) 

41 assert all([(isinstance(field, str) and field in var_info.index) for field in fields]) 

42 assert isinstance(y, str) 

43 assert y in var_info.index 

44 assert isinstance(file_loc, str) 

45 

46 # if x is numerical(continuous) field, we return the pearson correlation between a field and y 

47 # for the pearson correlation between a field and y, their size must be the same 

48 res = dict() 

49 res['continuous'] = dict() 

50 res['categorical'] = dict() 

51 for x in fields: 

52 if var_info.loc[x]['type'] == 'continuous': 

53 res['continuous'][x] = df[x].corr(df[y], method=method) 

54 elif var_info.loc[x]['type'] == 'categorical': 

55 df_sub = df[[x, y]].dropna() 

56 data = [x for _, x in df_sub.groupby(by=x)[y]] 

57 res['categorical'][x] = tuple(stats.f_oneway(*data)) 

58 

59 return res

105 def train_model(self, y, fields, regressor=None, test_split=0): 

106 """ 

107 train a machine learning model with y as dependent variable and variables in fields parameter as independent variables 

108 :param regressor: sklearn regressor object, if None default RandomForestRegressor is used 

109 :param test_split: float, if non-zero, train-test split is performed and training and test accuracy is returned else 

110 model trained on complete data and training accuracy along with -1 in place of test accuracy returned. 

111 :param y: string, dependent variable, should be numerical/continuous 

112 :param fields: list, list of independent variables, can be numerical or categorical 

113 :return: returns a tuple with training accuracy and test accuracy if test_split > 0 else a tuple with training accuracy 

114 and -1 in place of test accuracy. 

115 """ 

116 assert isinstance(fields, list) 

117 assert all([(isinstance(field, str) and field in self.var_info.index) for field in fields]) 

118 assert y in self.var_info.index 

119 assert 0 <= test_split <= 1 

120 

121 df_sub = self.df[fields + [y]] 

122 df_sub = df_sub.dropna() 

123 

124 self.fields = fields 

125 Y = df_sub[y] 

126 df_sub = df_sub[fields] 

127 

128 self.cat_cols = [field for field in fields if self.var_info.loc[field]['type'] == 'categorical'] 

129 self.cont_cols = [field for field in fields if self.var_info.loc[field]['type'] == 'continuous'] 

130 # print(f'Cat: {self.cat_cols} and cont: {self.cont_cols}') 

131 

132 numeric_transformer = Pipeline(steps=[ 

133 ('imputer', SimpleImputer(strategy='median')), 

134 ('scaler', StandardScaler())]) 

135 

136 categorical_transformer = Pipeline(steps=[ 

137 ('imputer', SimpleImputer(strategy='constant', fill_value='missing')), 

138 ('onehot', OneHotEncoder(handle_unknown='ignore'))]) 

139 

140 preprocessor = ColumnTransformer( 

141 transformers=[ 

142 ('num', numeric_transformer, self.cont_cols), 

143 ('cat', categorical_transformer, self.cat_cols)]) 

144 

145 # Append classifier to preprocessing pipeline. 

146 # Now we have a full prediction pipeline. 

147 from sklearn.ensemble import RandomForestRegressor 

148 from sklearn.model_selection import train_test_split 

149 if regressor is None: 

150 regressor = RandomForestRegressor() 

151 self.clf = Pipeline(steps=[('preprocessor', preprocessor), 

152 ('classifier', regressor)]) 

153 

154 if test_split > 0: 

155 X_train, X_test, y_train, y_test = train_test_split(df_sub, Y, test_size=0.2) 

156 self.clf.fit(X_train, y_train) 

157 self.trained = True 

158 return self.clf.score(X_train, y_train), self.clf.score(X_test, y_test) 

159 else: 

160 self.clf.fit(df_sub, Y) 

161 self.trained = True 

162 return self.clf.score(df_sub, Y), -1
